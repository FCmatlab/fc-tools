
\begin{itemize}
\item[$\bullet$] \fcmcode{fc_tools.sys.getComputerName()} returns the name of the computer as a string.
\item[$\bullet$] \fcmcode{fc_tools.sys.getUserName()} returns the name (login) of the current user as a string.
\item[$\bullet$] \fcmcode{fc_tools.sys.getRAM()} returns available memory (RAM) in GB of the computer.
\item[$\bullet$] \fcmcode{fc_tools.sys.getCPUinfo()} returns CPU(s) informations as a structure.
\item[$\bullet$] \fcmcode{fc_tools.sys.getOSinfo()} returns OS informations as a structure.
\item[$\bullet$] \fcmcode{fc_tools.sys.isfolder()} return true if a folder exists.
\item[$\bullet$] \fcmcode{fc_tools.sys.isfileexists()} return true if a file exists.
\end{itemize}
